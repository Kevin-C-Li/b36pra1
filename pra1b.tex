	\section*{Program}
	Precondition: \(n\in \mathbb{N} \) \\ 
	Postcondition: TRUE iff n is prime, FALSE otherwise\\
	is\_prime(n): \\
	\begin{verbatim}
1        i=2; prime = TRUE
2        while(prime and i<n):
3            if(n%i == 0):
4                prime = FALSE
5            else:
6                i = i+1
7        return prime
	\end{verbatim}
